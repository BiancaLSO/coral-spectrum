ColumnView Component: Navigation

Component API 

// parent class
.coral-ColumnView

// modifier for single column navigation (as seen in the rail).
.coral-ColumnView--navigation

// modifier for Miller columns.
.coral-ColumnView--miller

// each of the columns that are displayed.
.coral-ColumnView-column

// active column.
.coral-ColumnView-column.is-active

// each item inside the column.
.coral-ColumnView-item

// modifier for items with doublespaced labels (Miller columns).
// is applied dynamically if the normal label is overflowing.
// word wrapping can be avoided by data-wrapLabels="false".
.coral-ColumnView-item--doublespaced

// active item (navigation).
.coral-ColumnView-item.is-active

// selected item (Miller columns).
.coral-ColumnView-item.is-selected

// checked item (Miller columns).
.coral-ColumnView-item.is-selected.is-checked

// adds the > to the item to indicate it can be navigated.
.coral-ColumnView-item--hasChildren

// header item, used to separate the items.
.coral-ColumnView-item--header

// item that has back navigation.
.coral-ColumnView-item--back

// the icon (navigation).
// wrapper of the icon or the thumbnail (Miller columns).
.coral-ColumnView-icon

// the thumbnail (Miller columns).
.coral-ColumnView-thumbnail

// the preview column (Miller columns).
.coral-ColumnView-column.coral-ColumnView-preview

// wrapper of the icon or the thumbnail in a preview.
.coral-ColumnView-preview-icon

// the thumbnail in a preview.
.coral-ColumnView-preview-thumbnail

// a label in a preview.
.coral-ColumnView-preview-label

// a value (assigned to a label) in a preview.
.coral-ColumnView-preview-value
