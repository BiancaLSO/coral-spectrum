coral-Forms

Vertical Layout

/**
 * Indicates that the form is using vertical layout, where the label is placed above the field.
 */
.coral-Form.coral-Form--vertical

/**
 * Indicates the fieldset.
 * .fieldset is used instead of <fieldset> as <legend> cannot be styled as per requirement.
 */
.coral-Form--vertical .coral-Form-fieldset

/**
 * Indicates the legend of fieldset.
 */
.coral-Form--vertical .coral-Form-fieldset > .coral-Form-fieldset-legend

/**
 * Indicates the label of the field.
 * .fieldlabel is used so that the styling is only applied to this label, not any <label> inside the form.
 */
.coral-Form--vertical .coral-Form-fieldlabel

/**
 * Indicates the field.
 */
.coral-Form--vertical .coral-Form-field

/**
 * Indicates the field wrapper. A field wrapper MUST be used for field info and field error.
 */
.coral-Form--vertical .coral-Form-fieldwrapper

/**
 * Align a fieldwrapper right
 */
.coral-Form--vertical .coral-Form-fieldwrapper.coral-Form-fieldwrapper--alignRight

/**
 * Indicates the field info.
 */
.coral-Form--vertical .coral-Form-fieldwrapper .coral-Form-fieldinfo

/**
 * Indicates the field error.
 */
.coral-Form--vertical .coral-Form-fieldwrapper .coral-Form-fielderror
